# Requires Python 3.11 or 3.12 (we use 3.12, installed via `uv`).
# Do NOT use the system's Python 3.14 -- it is too new for parts of the ML
# stack (PyTorch/pyannote). The project venv is already pinned to CPython 3.12.
#
# --- Speech-to-text engine (sense_ear module) ---
# faster-whisper runs Whisper through CTranslate2 (no PyTorch needed).
faster-whisper==1.2.1
ctranslate2==4.8.0

# CUDA 12 runtime libraries.
# CTranslate2 4.8 is built against CUDA 12, but the system may ship a different
# CUDA toolkit (e.g. CUDA 13). These wheels provide libcublas.so.12 / libcudnn.so.9
# and, at 12.9 / 9.23, include kernels for Blackwell GPUs (RTX 50-series, sm_120).
# sense_ear preloads them at runtime, so no LD_LIBRARY_PATH tweaking is required.
nvidia-cublas-cu12==12.9.2.10
nvidia-cudnn-cu12==9.23.2.1

# --- Secrets ---
# Loads HF_TOKEN (and any other secret) from a git-ignored .env file.
python-dotenv==1.2.1

# --- Thinking engine (mind module) ---
# A minimal LangGraph graph over an OpenAI chat model: text in -> reply out.
langgraph==0.6.7
langchain-openai==0.3.33

# --- Text-to-speech engine (effectors.effector_voice module) ---
# The voice uses a design-once -> clone pattern: VoiceDesign bakes a reference
# clip once (scripts/bake_voice.py), and the Base model clones it at runtime.
# qwen-tts pulls in numpy/librosa; the effector's sentence streaming uses only
# the standard library (queue/threading/re), so it adds no new dependency.
# Pin numba away from old Python<3.10 builds that can be selected through the
# loose librosa dependency in qwen-tts.
qwen-tts==0.1.1
numba>=0.61
soundfile==0.14.0

# Optional: flash-attention 2 makes Qwen3-TTS faster. The effector requests it
# by default and falls back gracefully if it is absent, so this is NOT required.
# It is delicate to build on Blackwell GPUs, hence left commented out. To try it:
#   uv pip install flash-attn --no-build-isolation
# flash-attn

# --- Demo UI ---
streamlit==1.58.0
streamlit-mic-recorder==0.0.8
